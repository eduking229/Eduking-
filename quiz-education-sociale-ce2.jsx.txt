function shuffle(arr) {
  return [...arr].sort(() => Math.random() - 0.5);
}

  const [mode, setMode] = useState("accueil"); // accueil | choix | devinette | quiz | resultats
  const [seqFilter, setSeqFilter] = useState(null);
  const [questions, setQuestions] = useState([]);
  const [devinettes, setDevinettes] = useState([]);
  const [index, setIndex] = useState(0);
  const [score, setScore] = useState(0);
  const [selected, setSelected] = useState(null);
  const [showAnswer, setShowAnswer] = useState(false);

  const [devReponse, setDevReponse] = useState("");

  function demarrerQuiz(seqId) {
    const pool = seqId ? QUIZ.filter(q => q.seq === seqId) : shuffle(QUIZ).slice(0, 10);
    setQuestions(shuffle(pool));
    setIndex(0); setScore(0); setSelected(null); setShowAnswer(false);
    setTotal(pool.length);
    setMode("quiz");
  }

  function demarrerDevinette(seqId) {
    const pool = seqId ? DEVINETTES.filter(d => d.seq === seqId) : shuffle(DEVINETTES);
    setDevinettes(pool);
    setIndex(0); setScore(0); setDevReponse(""); setDevRevele(false); setShowIndice(false);
    setTotal(pool.length);
    setMode("devinette");
  }

    if (showAnswer) return;

    setShowAnswer(true);
    if (i === questions[index].correct) {
      setScore(s => s + 1);
      setCelebrate(true);
      setTimeout(() => setCelebrate(false), 800);
    }
  }

  function suivant() {
    if (index + 1 >= questions.length) {
      setMode("resultats");
    } else {
      setIndex(i => i + 1);
      setSelected(null);
      setShowAnswer(false);
    }
  }

  function suivantDev() {
    if (index + 1 >= devinettes.length) {
      setMode("resultats");
    } else {
      setIndex(i => i + 1);
      setDevReponse("");
      setDevRevele(false);
      setShowIndice(false);
    }
  }

  function revelerReponse() {
    setDevRevele(true);

          </button>
        )}
        <style>{`@keyframes pop{0%{opacity:0;transform:translate(-50%,-50%) scale(0.5)}50%{opacity:1;transform:translate(-50%,-50%) scale(1.3)}100%{opacity:0;transform:translate(-50%,-50%) scale(1)}}`}</style>
      </div>
    );
  }

  // JEU DEVINETTE
  if (mode === "devinette" && devinettes.length > 0) {
    const d = devinettes[index];
    const seq = SEQUENCES.find(s => s.id === d.seq);
    return (
      <div style={{minHeight:"100vh",background:"#0f0f23",padding:"16px",fontFamily:"'Segoe UI',sans-serif"}}>
        <div style={{display:"flex",justifyContent:"space-between",alignItems:"center",marginBottom:16}}>
          <button onClick={() => setMode("accueil")} style={{background:"none",border:"none",color:"#aaa",fontSize:20,cursor:"pointer"}}>✕</button>
          <span style={{color:"#aaa",fontSize:13}}>{index+1} / {total}</span>
          <div style={{background:"#FFD70022",border:"1px solid #FFD70066",borderRadius:20,padding:"4px 12px"}}>
            <span style={{color:"#FFD700",fontWeight:800,fontSize:14}}>⭐ {score}</span>
          </div>
        </div>

        <div style={{background:`${seq.couleur}22`,border:`1px solid ${seq.couleur}55`,borderRadius:8,padding:"6px 12px",display:"inline-flex",alignItems:"center",gap:6,marginBottom:18}}>
          <span>{seq.emoji}</span>
          <span style={{color:seq.couleur,fontSize:11,fontWeight:700}}>{seq.titre}</span>
        </div>

        {/* Devinette card */}
        <div style={{background:"linear-gradient(135deg,rgba(255,255,255,0.08),rgba(255,255,255,0.03))",borderRadius:20,padding:24,marginBottom:18,border:"1px solid rgba(255,255,255,0.12)",position:"relative"}}>
          <div style={{fontSize:36,marginBottom:12,textAlign:"center"}}>🤔</div>
          <p style={{color:"#fff",fontSize:15,fontWeight:600,lineHeight:1.6,margin:0,textAlign:"center",fontStyle:"italic"}}>
            « {d.texte} »
          </p>
        </div>

        {/* Indice */}
        {!devRevele && (
          <button onClick={() => setShowIndice(true)} style={{width:"100%",background:"rgba(253,203,110,0.1)",border:"1px solid rgba(253,203,110,0.4)",borderRadius:10,padding:"10px",color:"#FDCB6E",fontSize:13,fontWeight:600,cursor:"pointer",marginBottom:10}}>
            💡 Voir un indice
          </button>
        )}
        {showIndice && !devRevele && (
          <div style={{background:"rgba(253,203,110,0.1)",border:"1px solid #FDCB6E55",borderRadius:10,padding:"12px",marginBottom:12}}>
            <span style={{color:"#FDCB6E",fontSize:13}}>💡 <strong>Indice :</strong> {d.indice}</span>
          </div>
        )}

        {!devRevele ? (
          <button onClick={revelerReponse} style={{width:"100%",background:"linear-gradient(90deg,#00B894,#00CEC9)",border:"none",borderRadius:14,padding:"16px",color:"#fff",fontSize:15,fontWeight:800,cursor:"pointer",marginTop:4}}>
            🔓 Révéler la réponse
          </button>
        ) : (
          <div>
            <div style={{background:"rgba(0,184,148,0.15)",border:"2px solid #00B894",borderRadius:16,padding:"18px",textAlign:"center",marginBottom:14}}>
              <div style={{color:"#aaa",fontSize:11,marginBottom:6}}>RÉPONSE :</div>
              <div style={{color:"#00B894",fontSize:20,fontWeight:900}}>{d.reponse}</div>
            </div>
            <button onClick={suivantDev} style={{width:"100%",background:"linear-gradient(90deg,#5F27CD,#54A0FF)",border:"none",borderRadius:14,padding:"16px",color:"#fff",fontSize:16,fontWeight:800,cursor:"pointer"}}>
              {index+1 >= devinettes.length ? "Voir mes résultats 🏆" : "Devinette suivante →"}
            </button>
          </div>
        )}
      </div>
    );
  }

  // RÉSULTATS
  if (mode === "resultats") {
    const pct = Math.round((score / total) * 100);
    const mention = pct >= 80 ? { t:"Excellent ! 🏆", c:"#FFD700" } : pct >= 60 ? { t:"Bien ! 👍", c:"#00B894" } : pct >= 40 ? { t:"Passable 😊", c:"#FF9F43" } : { t:"À réviser 📖", c:"#FF6B6B" };
    return (
      <div style={{minHeight:"100vh",background:"#0f0f23",display:"flex",flexDirection:"column",alignItems:"center",justifyContent:"center",padding:"24px",fontFamily:"'Segoe UI',sans-serif"}}>
        <div style={{fontSize:72,marginBottom:16}}>{pct >= 80 ? "🏆" : pct >= 60 ? "🎉" : pct >= 40 ? "😊" : "📖"}</div>
        <h2 style={{color:mention.c,fontSize:26,fontWeight:900,margin:"0 0 6px"}}>{mention.t}</h2>
        <p style={{color:"#aaa",fontSize:14,marginBottom:24}}>Tu as répondu correctement à</p>

        <div style={{background:"rgba(255,255,255,0.05)",border:"2px solid rgba(255,255,255,0.15)",borderRadius:24,padding:"32px 48px",textAlign:"center",marginBottom:28}}>
          <div style={{fontSize:52,fontWeight:900,color:mention.c}}>{score}<span style={{fontSize:28,color:"#555"}}>/{total}</span></div>
          <div style={{color:"#aaa",fontSize:14,marginTop:4}}>{pct}% de réussite</div>
        </div>

        {/* Barre */}
        <div style={{width:"100%",maxWidth:280,background:"#1a1a2e",borderRadius:20,height:14,overflow:"hidden",marginBottom:32}}>
          <div style={{width:`${pct}%`,height:"100%",background:`linear-gradient(90deg,${mention.c},${mention.c}99)`,borderRadius:20,transition:"width 1s"}} />
        </div>

        <div style={{display:"flex",flexDirection:"column",gap:12,width:"100%",maxWidth:300}}>
          <button onClick={() => setMode("accueil")} style={{background:"linear-gradient(90deg,#5F27CD,#54A0FF)",border:"none",borderRadius:14,padding:"16px",color:"#fff",fontSize:15,fontWeight:800,cursor:"pointer"}}>
            🏠 Retour à l'accueil
          </button>
          <button onClick={() => { setMode("accueil"); setTimeout(() => demarrerQuiz(null), 100); }} style={{background:"rgba(255,255,255,0.06)",border:"1px solid rgba(255,255,255,0.15)",borderRadius:14,padding:"16px",color:"#fff",fontSize:14,fontWeight:600,cursor:"pointer"}}>
            🔄 Rejouer
          </button>
        </div>
      </div>
    );
  }

  return null;
}
